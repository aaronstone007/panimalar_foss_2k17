Control Flow

If-else


		if conditional [then]
		  code...
		[elsif conditional [then]
		   code...]...
		[else
		   code...]
		end
					

		x = 1
		if x > 2
		   puts "x is greater than 2"
		elsif x <= 2 and x!=0
		   puts "x is 1"
		else
		   puts "I can't guess the number"
		end

Unless


		unless conditional [then]
		   code
		[else
		   code ]
		end
					

		x = 1
		unless x>=2
		   puts "x is less than 2"
		else
		   puts "x is greater than 2"
		end

Case


		case expression
		[when expression [, expression ...] [then]
		   code ]...
		[else
		   code ]
		end

		when 7 .. 12
		   puts "child"
		when 13 .. 18

		end
